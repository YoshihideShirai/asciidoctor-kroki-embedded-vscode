:kroki-default-format: svg

== Architecture

[c4plantuml]
----
@startuml
!include <C4/C4_Container>

Person(author, "Author")
System_Boundary(vscode, "VS Code") {
  Container(extension, "Kroki Embedded Extension", "TypeScript")
  Container(preview, "AsciiDoc Preview", "Webview")
}
System_Ext(kroki, "Kroki renderer")

Rel(author, extension, "writes AsciiDoc")
Rel(extension, preview, "registers diagrams")
Rel(preview, kroki, "hydrates embedded SVG")
@enduml
----

[graphviz]
----
digraph G {
  rankdir=LR
  node [shape=box, style=rounded]
  adoc [label="AsciiDoc"]
  extension [label="VS Code extension"]
  preview [label="Preview"]
  svg [label="Embedded SVG"]

  adoc -> extension -> preview -> svg
}
----

[blockdiag]
----
blockdiag {
  default_shape = roundedbox;
  orientation = portrait;

  source -> register -> preview -> hydrate -> render;
}
----
